create database student 

create table student(
sno char(9) not null primary key,
sname char(10),
sage int,
ssex char(2),
sdept char(20),
)

create table course(
cno char(9) not null primary key,
cname char(10),
cpno char(4),
ccredit int,
)

create table sc(
sno char(9) not null,
cno char(9) not null,
grade int,
primary key(sno,cno),
foreign key (sno) references student(sno),
foreign key (cno) references course(cno),
)
